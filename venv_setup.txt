# 1. Create the virtual environment
py -3 -m venv venv

# IDE Configuration: 
# In VS Code Command Palette (Ctrl+Shift+P), select the following as your Python Interpreter:
.\venv\Scripts\python.exe

# 2. Activate the virtual environment
# For Windows Command Prompt:
venv\Scripts\activate.bat
# For Windows PowerShell:
# .\venv\Scripts\Activate.ps1

# 3. Install FastAPI and all optional dependencies
pip install fastapi[all]

# 4.save installed packages to requirements.txt
pip freeze > requirements.txt


# 5. Verify the installed packages
pip list

# 6. Run the application
# Standard Dev Mode (Recommended):
fastapi dev

# Manual Uvicorn Mode:
# uvicorn main:app --reload
